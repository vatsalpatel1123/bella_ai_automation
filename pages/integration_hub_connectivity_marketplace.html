// Example API request to create a new lead
const response = await fetch('https://api.bellaai.com/v1/leads', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'Authorization': 'Bearer YOUR_API_KEY'
  },
  body: JSON.stringify({
    firstName: 'Jane',
    lastName: 'Smith',
    email: 'jane.smith@example.com',
    phone: '+1234567890',
    source: 'Website Contact Form',
    interests: ['AI Automation', 'Lead Generation'],
    customFields: {
      company: 'Acme Inc',
      role: 'Marketing Director'
    }
  })
});

const data = await response.json();
console.log('New lead created:', data.id);
